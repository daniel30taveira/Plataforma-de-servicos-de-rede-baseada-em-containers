\section{Docker Compose da Plataforma (\texttt{docker-compose.yml})}
\begin{lstlisting}[language=bash, frame=single, breaklines=true, basicstyle=\footnotesize\ttfamily, extendedchars=false]
version: '3.8'

services:
  # REVERSE PROXY E SERVIDOR WEB (Nginx)
  proxy_web:
    image: nginx:latest
    container_name: nginx_proxy
    restart: unless-stopped
    ports:
      - "80:80"
      - "443:443"
    volumes:
      - ./web:/usr/share/nginx/html:ro
      - ./nginx/default.conf:/etc/nginx/conf.d/default.conf:ro
    networks:
      - rede_interna

  # SERVIDOR DNS (Pi-Hole)
  servidor_dns:
    image: pihole/pihole:latest
    container_name: dns_pihole
    restart: unless-stopped
    ports:
      - "53:53/tcp"
      - "53:53/udp"
      - "8080:80"
    environment:
      TZ: 'Europe/Lisbon'
      WEBPASSWORD: "{{ vault_pihole_password }}"
    networks:
      - rede_interna

  # REDE PRIVADA VIRTUAL (WireGuard VPN)
  wireguard:
    image: lscr.io/linuxserver/wireguard:latest
    container_name: wireguard
    cap_add:
      - NET_ADMIN
      - SYS_MODULE
    environment:
      - PUID=1000
      - PGID=1000
      - TZ=Europe/Lisbon
      - SERVERURL=daniel-infra.ddns.net
      - SERVERPORT=51821
      - PEERS=3 
      - PEERDNS=192.168.1.252
      - INTERNAL_SUBNET=10.13.13.0
      - ALLOWEDIPS=0.0.0.0/0
    volumes:
      - ./wireguard:/config
    ports:
      - "51821:51821/udp"
    sysctls:
      - net.ipv4.conf.all.src_valid_mark=1
    restart: unless-stopped
    networks:
      - rede_interna

  # SERVIDOR DE FICHEIROS / CLOUD PRIVADA (Nextcloud)
  nextcloud:
    image: lscr.io/linuxserver/nextcloud:latest
    container_name: nextcloud
    environment:
      - PUID=1000
      - PGID=1000
      - TZ=Europe/Lisbon
    volumes:
      - ./nextcloud/appdata:/config
      - ./nextcloud/data:/data
    ports:
      - "8081:443" 
    restart: unless-stopped
    networks:
      - rede_interna

  # SERVIDOR DE E-MAIL (Docker-Mailserver)
  mailserver:
    image: mailserver/docker-mailserver:latest
    container_name: mailserver
    hostname: mail
    domainname: local.lan
    ports:
      - "25:25"
      - "143:143"
      - "587:587"
      - "993:993"
    volumes:
      - ./mailserver/maildata:/var/mail
      - ./mailserver/mailstate:/var/mail-state
      - ./mailserver/mailconfig:/tmp/docker-mailserver
      - /etc/localtime:/etc/localtime:ro
    environment:
      - ENABLE_SPAMASSASSIN=1
      - ENABLE_CLAMAV=0
      - ENABLE_FAIL2BAN=1
      - ONE_DIR=1
    restart: unless-stopped
    networks:
      - rede_interna

  # Node Exporter (Metricas do Host)
  node-exporter:
    image: prom/node-exporter:latest
    container_name: node_exporter
    restart: unless-stopped
    ports:
      - "9100:9100"
    networks:
      - rede_interna

  # Prometheus (Base de dados de metricas)
  prometheus:
    image: prom/prometheus:latest
    container_name: prometheus
    restart: unless-stopped
    ports:
      - "9090:9090"
    volumes:
      - ./prometheus/prometheus.yml:/etc/prometheus/prometheus.yml:ro
    networks:
      - rede_interna

  # Grafana (Visualizacao de Dashboards)
  grafana:
    image: grafana/grafana:latest
    container_name: grafana
    restart: unless-stopped
    ports:
      - "3000:3000"
    environment:
      - GF_SECURITY_ADMIN_PASSWORD={{ vault_grafana_password }}
      - GF_SERVER_ROOT_URL=http://192.168.1.252/grafana/
      - GF_SERVER_SERVE_FROM_SUB_PATH=true
    volumes:
      - ./grafana/provisioning:/etc/grafana/provisioning
    networks:
      - rede_interna

  # Grafana Loki (Agregador de Logs)
  loki:
    image: grafana/loki:3.0.0
    container_name: loki
    ports:
      - "3100:3100"
    restart: unless-stopped
    networks:
      - rede_interna

  # Promtail (Agente de recolha de logs)
  promtail:
    image: grafana/promtail:3.0.0
    container_name: promtail
    volumes:
      - /var/log:/var/log:ro
      - /var/lib/docker/containers:/var/lib/docker/containers:ro
      - /var/run/docker.sock:/var/run/docker.sock:ro
      - ./promtail/promtail-config.yaml:/etc/promtail/config.yml:ro
    command: -config.file=/etc/promtail/config.yml
    restart: unless-stopped
    networks:
      - rede_interna

networks:
  rede_interna:
    driver: bridge
\end{lstlisting}

\section{Ansible Playbook de Setup (\texttt{setup\_docker.yml})}
\begin{lstlisting}[language=bash, frame=single, breaklines=true, basicstyle=\footnotesize\ttfamily, extendedchars=false]
---
- name: Configuracao inicial do servidor e instalacao do Docker
  hosts: servidores
  become: yes

  tasks:
    - name: Atualizar a cache do apt
      apt:
        update_cache: yes
        cache_valid_time: 3600

    - name: Instalar pacotes pre-requisitos do sistema
      apt:
        name:
          - apt-transport-https
          - ca-certificates
          - curl
          - software-properties-common
        state: present

    - name: Instalar o Docker e o Docker Compose
      apt:
        name:
          - docker.io
          - docker-compose-v2
        state: present

    - name: Garantir que o servico docker arranca
      systemd:
        name: docker
        state: started
        enabled: yes

    - name: Adicionar o utilizador ao grupo docker
      user:
        name: "{{ ansible_user }}"
        groups: docker
        append: yes
\end{lstlisting}

\section{Ansible Playbook de Deploy (\texttt{deploy\_servicos.yml})}
\begin{lstlisting}[language=bash, frame=single, breaklines=true, basicstyle=\footnotesize\ttfamily, extendedchars=false]
---
- name: Deploy da Stack de Servicos de Rede
  hosts: localhost
  connection: local
  become: yes

  vars_files:
    - vars/secrets.yml 

  tasks:
    - name: Garantir que a pasta de destino existe
      ansible.builtin.file:
        path: /opt/plataforma_servicos
        state: directory
        mode: '0755'

    - name: Copiar pastas de configuracao essenciais
      ansible.builtin.copy:
        src: "../servicos/{{ item }}"
        dest: /opt/plataforma_servicos/
        mode: '0755'
      loop:
        - nginx
        - prometheus
        - promtail
        - grafana
        - web

    - name: Copiar e configurar o docker-compose com Vault
      ansible.builtin.template:
        src: ../servicos/docker-compose.yml
        dest: /opt/plataforma_servicos/docker-compose.yml
        mode: '0600' 

    - name: Executar o Docker Compose
      community.docker.docker_compose_v2:
        project_src: /opt/plataforma_servicos
        state: present
        pull: missing
\end{lstlisting}

\section{GitLab CI/CD Pipeline (\texttt{.gitlab-ci.yml})}
\begin{lstlisting}[language=bash, frame=single, breaklines=true, basicstyle=\footnotesize\ttfamily, extendedchars=false]
stages:
  - deploy
 
deploy_infraestrutura:
  stage: deploy
  tags:
    - meu-runner-local
  script:
    - echo "A iniciar o deploy com segredos Vault..."
    - cd infra
    - echo "$ANSIBLE_VAULT_PASSWORD" > .vault_key
    - sudo ansible-playbook -i "localhost," -c local --vault-password-file .vault_key deploy_servicos.yml
    - rm -f .vault_key
    - echo "Deploy concluido com sucesso!"
\end{lstlisting}
